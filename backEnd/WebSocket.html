# WebSocket

# 什么是WebSocket?

WebSocekt技术是一种服务器推送技术，它的最大特点是服务器可以主动向客户端推送信息，客户端也可以向服务端发送消息，是真正的双向平等通信。

# 如何实现WebSocket通信?

WebSocket实现分为服务端实现和客户端实现，两者配置都正常时，才能连通。

# 服务端实现过程

服务端是由java语言编写的，项目基于springBoot。

# pom文件中引入包

<dependency>

</dependency>

# 服务端（java实现）完整代码

package com.xueliman.iov.biz.webdba.web.websocket;

import java.io.IOException;
import java.util.Map;
import java.util.Map.Entry;
import java.util.concurrent.ConcurrentHashMap;

import javax.websocket.OnClose;
import javax.websocket.OnError;
import javax.websocket.OnMessage;
import javax.websocket.OnOpen;
import javax.websocket.Session;
import javax.websocket.server.PathParam;
import javax.websocket.server.ServerEndpoint;

import org.apache.log4j.Logger;
import org.springframework.stereotype.Component;

import jline.internal.Log;

/**
 * 警报websocket 
 * @author yjw
 *
 */
@ServerEndpoint("/alarmSocket")
@Component
public class AlarmWebSocket {

	private static Logger logger = Logger.getLogger(AlarmWebSocket.class.getName());


	/**
	 * 用于存储 webSocket session
	 */
	private static Map<String, Session> webSocketSessions = new ConcurrentHashMap<>();

	public AlarmWebSocket() {
		// TODO Auto-generated constructor stub
	}

	/**
	 * open方法,建立连接时触发
	 * @param session
	 * @param userid
	 */
	@OnOpen
	public void open(Session session) {
		webSocketSessions.put("heart_" + session.getId(), session);
		logger.info("session：" + session.getId() + "  ---建立连接,当前连接数量：" + webSocketSessions.size());
	}

	/**
	 * message方法
	 * @param session
	 * @param message
	 */
	@OnMessage
	public void message(Session session, String message) {
		logger.info("来自session" + session.getId() + "的消息:" + message);
		if ("HeartBeat".equals(message)) {
			try {
				session.getBasicRemote().sendText("success");
			} catch (Exception e) {
				logger.error(e.getMessage());
			}
		}
	}

	/**
	 * 向客户端推送信息的方法
	 * 
	 * @param message
	 *            要推送的信息
	 */
    public static synchronized void sendMessage(String message) throws IOException {
        for (Entry<String, Session> entry : webSocketSessions.entrySet()) {
            Session sess = (Session) entry.getValue();
            try {
                sess.getBasicRemote().sendText(message);
                logger.info("向session:" + sess.getId() + "发送message： " + message);
            } catch (Exception e) {
                Log.error(e.getMessage() + ":sessionId" + sess.getId());
            }
        }
    }

	/**
	 * close方法,session断开时触发
	 * @param session
	 */
	@OnClose
	public void close(Session session) {
	    webSocketSessions.remove("heart_" + session.getId());
	    logger.info("session：" + session.getId() + "  ---建立连接,当前连接数量：" + webSocketSessions.size());
	}
	
	/**
	 * error方法
	 * @param session
	 * @param error
	 */
	@OnError
	public void onError(Session session, Throwable error) {
		logger.error(error.getMessage());
		close(session);
	}
}

# 客户端（vue实现）完整代码

<template>
    <div>测试数据为：{{ readData }}</div>
</template>

<script>
export default {
  data () {
    var vm = this
    return {
      readData: 2,
      websocket: null,
      websocketUrl: 'ws://192.168.1.74:8080/webdba/alarmSocket', // websocket 链接url
      lockReconnect: false, // 避免重复连接
      heartCheck: {
        websocket_timeout: 60000, // websocket 超时时间
        websocket_timeoutObj: null, // websocket 定时器对象
        websocket_serverTimeoutObj: null,
        // 重连方法
        reset: function () {
          var that = this
          clearTimeout(that.timeoutObj)
          clearTimeout(that.serverTimeoutObj)
          return this
        },
        start: function () {
          var self = this
          this.timeoutObj = setTimeout(function () {
            vm.websocket.send('HeartBeat')
            self.serverTimeoutObj = setTimeout(function () {
              vm.websocket.close()// 如果onclose会执行reconnect，我们执行ws.close()就行了.如果直接执行reconnect 会触发onclose导致重连两次
            }, self.websocket_timeout)
          }, this.websocket_timeout)
        }
      }
    }
  },
  created () {
    this.createWebSocket()
  },
  destroyed: function () {
    this.websocket.onclose()
    // 页面销毁时关闭长连接
  },
  methods: {
    // 实际调用方法
    createWebSocket: function () {
      var that = this
      try {
        that.websocket = new WebSocket(that.websocketUrl)
        that.initEventHandle()
      } catch (e) {
        that.websocketreconnect(that.url)
      }
    },

    // init方法
    initEventHandle: function () {
      var that = this
      // open方法
      that.websocket.onopen = function () {
        that.heartCheck.reset().start()
      }
      // onmessage 方法
      that.websocket.onmessage = function (data) {
        if (data.data == 'success') {
          that.heartCheck.reset().start()
        } else {
          console.log(data.data)
        }
      }
      // onclose 方法
      that.websocket.onclose = function () {
        that.websocketreconnect(that.url)
      }
    },

    // socket 重连方法
    websocketreconnect: function (url) {
      var that = this
      if (that.lockReconnect) {
        return
      }
      that.lockReconnect = true
      // 没连接上会一直重连，设置延迟避免请求过多
      setTimeout(function () {
        that.createWebSocket(that.websocketUrl)
        that.lockReconnect = false
      }, 2000)
    }
  }
}

</script>
